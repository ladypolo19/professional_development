function findLongestWordLength(str) {
  let word = str.split(' '); //para separarlo por palabra
  let longestWord = 0; //inicializa en cero
    for (let i = 0; i < word.length; i++) { //recorre cada palabra
      if (word[i].length > longestWord) { //con el index va comparando si es mayor a longestWord
        longestWord = word[i].length;
      }
    }
  return longestWord;
}
