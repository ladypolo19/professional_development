function findElement(arr, func) {
  let num = 0;

  for (let i = 0; i < arr.length; i++) { //loop para que recorra
    num = arr[i]; //cada indice
    if (func(num)) {
      return num;
    }
  }

  return undefined;
}

findElement([1, 2, 3, 4], num => num % 2 === 0);
