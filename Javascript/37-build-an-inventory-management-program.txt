let inventory = [];

function findProductIndex(productName) {
  //minimiza el nombre producto
  const searchName = productName.toLowerCase(); 
  //busca para dar el index
  return inventory.findIndex(product => 
    product.name.toLowerCase() === searchName
  );
}

function addProduct(productObject) {
  //minimiza el nombre del producto para coincidir inventario
  productObject.name = productObject.name.toLowerCase();
  //encuentra el indice
   let productIndex = findProductIndex(productObject.name);

   if(productIndex === -1) {
     //sino encuentra el indice lo crea en el array
     inventory.push(productObject);
     console.log(productObject.name + " " + "added to inventory");
   }
   else {
     inventory[productIndex].quantity += productObject.quantity;
     console.log(productObject.name + " " + "quantity updated");
   }
}

function removeProduct(productName, quantity) {
  //minimiza el nombre
  const normalizedName = productName.toLowerCase();
  const productIndex = findProductIndex(normalizedName);
  
  //Si no encuentra el producto
  if (productIndex === -1) {
    console.log(`${normalizedName} not found`);
    return;
  } 

  const currentProduct = inventory[productIndex];
  if (currentProduct.quantity < quantity) {
    console.log(`Not enough ${normalizedName} available, remaining pieces: ${currentProduct.quantity}`);
    
  } else {
    currentProduct.quantity -= quantity; 
    if (currentProduct.quantity === 0) {
      inventory.splice(productIndex, 1); 
    }
    console.log(`Remaining ${normalizedName} pieces: ${currentProduct.quantity}`);
  }
}
