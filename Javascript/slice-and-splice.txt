function frankenSplice(arr1, arr2, n) {
  let array = arr2.slice(); //replica de arr2

  for (let i=0; i < arr1.length; i++) {
    array.splice(n, 0, arr1[i]); // no elimina ningun elemento
    n++; //inserta elemento luego n++ siguiente
  }
  return array;
}

frankenSplice([1, 2, 3], [4, 5, 6], 1);
